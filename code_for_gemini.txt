    <script>
        // 기존의 script 로직(localStorage 저장, 계산 등)은 그대로 유지하시면 됩니다.
        // (글자수 제한상 로직은 이전과 동일하게 사용하세요)
        let ledgerData = JSON.parse(localStorage.getItem('bongRoadmap_v4')) || [];

        function toggleView(viewId) {
            document.getElementById('dashboard-view').style.display = viewId === 'dashboard-view' ? 'block' : 'none';
            document.getElementById('input-view').style.display = viewId === 'input-view' ? 'block' : 'none';
            if (viewId === 'input-view') initInputView();
            else renderTable();
        }

        function initInputView() {
            document.getElementById('income-list').innerHTML = '';
            document.getElementById('expense-list').innerHTML = '';
            document.getElementById('savings-list').innerHTML = '';
            addRow('income', '급여');
            addRow('expense', '식비');
            addRow('savings', '청약');
            calculateTotals();
        }

        function addRow(type, defaultName = '') {
            const list = document.getElementById(`${type}-list`);
            const row = document.createElement('div');
            row.className = 'item-row';
            row.innerHTML = `
                <input type="text" placeholder="항목명" value="${defaultName}" style="flex:1;">
                <div class="amount-wrapper">
                    <input type="text" placeholder="0" class="money-input item-amount">
                    <span class="unit">원</span>
                </div>
                <button class="btn-icon" onclick="this.parentElement.remove(); calculateTotals();">×</button>
            `;
            list.appendChild(row);

            row.querySelector('.money-input').addEventListener('input', function(e) {
                let value = e.target.value.replace(/,/g, '').replace(/[^0-9]/g, '');
                e.target.value = value ? Number(value).toLocaleString('ko-KR') : '';
                calculateTotals();
            });
        }

        function calculateTotals() {
            ['income', 'expense', 'savings'].forEach(type => {
                let sum = 0;
                document.querySelectorAll(`#${type}-list .item-amount`).forEach(input => {
                    sum += Number(input.value.replace(/,/g, '')) || 0;
                });
                document.getElementById(`total-${type}`).innerText = sum.toLocaleString() + '원';
            });
        }

        function saveData() {
            const month = document.getElementById('month-picker').value;
            if (!month) return alert("월을 선택해주세요.");

            const getSum = (type) => {
                let s = 0;
                document.querySelectorAll(`#${type}-list .item-amount`).forEach(i => s += Number(i.value.replace(/,/g, '')) || 0);
                return s;
            };

            ledgerData.push({ month, income: getSum('income'), expense: getSum('expense'), savings: getSum('savings') });
            ledgerData.sort((a, b) => a.month.localeCompare(b.month));
            localStorage.setItem('bongRoadmap_v4', JSON.stringify(ledgerData));
            toggleView('dashboard-view');
        }

        function renderTable() {
            const tbody = document.getElementById('dataTable');
            tbody.innerHTML = '';
            let prevBalance = 0;

            ledgerData.forEach((data, index) => {
                const currentMonthBalance = data.income - data.expense - data.savings;
                const cumulativeBalance = prevBalance + currentMonthBalance;

                tbody.innerHTML += `
                    <tr>
                        <td><strong>${data.month}</strong></td>
                        <td style="color: #adb5bd;">${prevBalance.toLocaleString()}원</td>
                        <td style="color: var(--primary); font-weight:600;">${data.income.toLocaleString()}원</td>
                        <td style="color: var(--danger); font-weight:600;">${data.expense.toLocaleString()}원</td>
                        <td>${data.savings.toLocaleString()}원</td>
                        <td>${currentMonthBalance.toLocaleString()}원</td>
                        <td class="total-asset">${cumulativeBalance.toLocaleString()}원</td>
                        <td><button class="btn btn-outline" style="padding:4px 8px; font-size:0.8em;" onclick="deleteData(${index})">삭제</button></td>
                    </tr>
                `;
                prevBalance = cumulativeBalance;
            });
        }

        function deleteData(index) {
            if(confirm("삭제하시겠습니까?")) {
                ledgerData.splice(index, 1);
                localStorage.setItem('bongRoadmap_v4', JSON.stringify(ledgerData));
                renderTable();
            }
        }

        window.onload = renderTable;
    </script>
